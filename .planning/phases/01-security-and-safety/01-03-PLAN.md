**SEC-01: Path traversal fix**

1. Add a minimal percent-decode helper function at the top of server.c (static, not exported). This is needed because `%2e%2e%2f` bypasses traversal checks without decoding:
```c
/* Minimal percent-decode: decodes %XX sequences in-place.
   Returns decoded length. Does NOT handle '+' as space. */
static size_t percent_decode(char *dst, const char *src, size_t src_len) {
    size_t j = 0;
    for (size_t i = 0; i < src_len; i++) {
        if (src[i] == '%' && i + 2 < src_len
            && isxdigit((unsigned char)src[i+1])
            && isxdigit((unsigned char)src[i+2])) {
            char hex[3] = { src[i+1], src[i+2], '\0' };
            dst[j++] = (char)strtol(hex, NULL, 16);
            i += 2;
        } else {
            dst[j++] = src[i];
        }
    }
    dst[j] = '\0';
    return j;
}
```
